async function toolsRunReels() {
  /* 0. Preflight: ffmpeg есть, транскрипт есть */
  var ff = AudioPreprocess.findFfmpegPath();
  if (!ff) return toolsShowErr('Нужен ffmpeg (для кадров vision и рендера субтитров).');
  var snap = await bridgeSnapshot();                     /* существующий паттерн getSnapshot */
  var seqName = snap.sequenceName;
  var tr = ContextStore.findTranscriptEntry(TRANSCRIPT_PID, seqName);
  if (!tr || !tr.entry || !tr.entry.segments || !tr.entry.segments.length)
    return toolsShowErr('Нет транскрипта для «' + seqName + '» — сначала транскрибируйте.');

  var fmt = document.getElementById('rl-format').value;          /* '9x16'|'1x1' */
  var targetW = 1080, targetH = fmt === '1x1' ? 1080 : 1920;
  var settings = ContextStore.getResolvedSettings();

  /* 1. Источники рефрейма (host) + размеры (ffprobe) — как в старом toolsRunVertical */
  var src = await pBridge('getVerticalReframeSources');
  /* dims: nestDims + probeVideoDimensions по уникальным mediaPath */

  /* 2. Смещения: ручные > vision > центр */
  var manual = DeterministicPipelines.parseVerticalOffsets(document.getElementById('rl-offsets').value);
  var visionNotes = [];
  var vp = ReelsPipeline.visionPlan(src.clips);
  var visionOffsets = [];
  for (var p of vp.paths) {
    /* пропустить, если имя клипа покрыто ручным смещением */
    try {
      var dur = await AudioPreprocess.probeDurationSec(p);
      var frame = await AudioPreprocess.extractFrameJpeg(p, dur / 2, { maxWidth: 768 });
      var resp = await CloudRuClient.chatCompletions({
        baseUrl: settings.baseUrl, apiKey: settings.apiKey,
        model: settings.visionModel || 'MiniMaxAI/MiniMax-M3',
        enableThinking: false, temperature: 0,
        responseFormat: 'json_object',
        messages: [{ role: 'user', content: [
          { type: 'text', text: 'Найди главного человека (лицо) в кадре. Ответ строго JSON: {"cx": <0..1 — горизонтальный центр лица, 0=левый край, 1=правый>}. Если людей нет: {"cx": 0.5}.' },
          { type: 'image_url', image_url: { url: frame } }] }]
      });
      var cx = JSON.parse(resp.choices[0].message.content).cx;
      var pct = ReelsPipeline.offsetPctFromCx(cx);
      if (pct === null) throw new Error('невалидный cx');
      /* смещение применяется ко всем клипам с этим mediaPath: используем match по имени файла */
      visionOffsets.push({ match: <имя файла без расширения, lowercase>, offsetPct: pct });
      visionNotes.push(имяФайла + ': vision ' + pct + '%');
    } catch (e) { visionNotes.push(имяФайла + ': vision не сработал (' + e.message + ') — центр'); }
  }
  var offsets = (manual || []).concat(visionOffsets);    /* manual раньше в массиве → substring-матч найдёт его первым */

  /* 3. Рефрейм (существующий путь) */
  var plan = DeterministicPipelines.planVerticalReframe(src.clips, dims, { targetW, targetH, offsets: offsets.length ? offsets : null });
  var applied = await pBridge('applyVerticalReframe', { expectedSequenceName: seqName, newName: seqName + ' — Рилс ' + (fmt === '1x1' ? '1x1' : '9x16'), targetW, targetH, items: plan.items });
  if (!applied.ok) return toolsShowErr(...);

  /* 4. Караоке-кьюи */
  var cues = ReelsPipeline.buildKaraokeCues(tr.entry.segments, {});

  /* 5. LLM-корректура: батчи по 40, chat-модель, JSON {"cues":[{"i":..,"text":..}]}; фейл → исходные */
  var proof = { applied: 0, rejected: 0 };
  try { proof = await reelsProofread(cues, settings); cues = proof.cues; }
  catch (e) { visionNotes.push('Корректура LLM не сработала: ' + e.message); }
  cues = cues.map(c => Object.assign({}, c, { text: ReelsPipeline.stripCueFinalPeriod(c.text) }));

  /* 6. ASS + рендер (fps из snap.timebase/секвенции; длительность = endSec последнего кью + 0.5) */
  var ass = ReelsPipeline.buildAss(cues, { w: targetW, h: targetH, fontName: <из rl-font>, textColor: <rl-text-color>, hlColor: <rl-hl-color>, anim: <rl-anim> });
  /* ~/.extensions_llm_chat_pr/reels/<seq>_<ts>.ass и .mov; fs.writeFileSync ass UTF-8 БЕЗ BOM */
  await runFfmpegArgs(ReelsPipeline.buildOverlayFfmpegArgs({ assPath, w: targetW, h: targetH, fps, durationSec, outPath }));

  /* 7. Вставка оверлея в НОВУЮ секвенцию: она стала активной после applyVerticalReframe */
  var ins = await pBridge('importAndOverlayOnTop', { filePath: outPath, expectedSequenceName: applied.sequenceName });
  if (!ins.ok) return toolsShowErr('Оверлей не вставлен: ' + ins.error);

  /* 8. Отчёт: секвенция, visionNotes, титров N, правок LLM applied/rejected */
}
